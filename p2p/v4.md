```
	func Encode(priv *ecdsa.PrivateKey, req Packet) (packet, hash []byte, err error) {
		b := new(bytes.Buffer)
		b.Write(headSpace)        // 97 empty bytes
		b.WriteByte(req.Kind())
		packet = b.Bytes()								// 97
		sig, err := crypto.Sign(crypto.Keccak256(packet[headSize:]), priv)

		copy(packet[macSize:], sig)
		hash = crypto.Keccak256(packet[macSize:])
		copy(packet, hash)
		return packet, hash, nil
} 
```
## Format 
- 0 - 31	32 bytes	MDC (Hash)	keccak256 de tout ce qui suit (Signature + Payload).
- 32 - 96	65 bytes	Signature	Signature ECDSA (r, s, v) du hash du reste du paquet.
- 97	1 byte	Packet Type	Identifie le message (ex: 0x01 pour Ping, 0x03 pour FindNode).
- 98 - Fin	Variable	Payload	Les données encodées en RLP (Recursive Length Prefix).
